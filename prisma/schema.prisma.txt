generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
}

model User {
  id        Int      @id @default(autoincrement())
  uid       String   @unique @default(uuid(7)) @db.Uuid
  name      String
  email     String   @unique
  password  String
  role      String   @default("user")
  status    Int      @default(1)
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")
  createdBy Int?     @map("created_by")
  updatedBy Int?     @map("updated_by")

  // Relações
  ownedProjects  Project[]      @relation("Owner")
  collaborators  Collaborator[] @relation("CollaboratorUser")
  assetsOwnerId  Asset[]        @relation("AssetOwner")
  invitesCreated Invite[]       @relation("InviteCreator")
  invitesUsed    Invite[]       @relation("InviteUsedBy")

  @@map("users")
}

model Project {
  id        Int      @id @default(autoincrement())
  uid       String   @unique @default(uuid(7)) @db.Uuid
  name      String   @unique
  ownerId   Int      @map("owner_id")
  owner     User     @relation("Owner", fields: [ownerId], references: [id])
  status    Int      @default(1) // 1 = ativo, -1 = excluído
  createdAt DateTime @default(now()) @map("created_at")
  createdBy Int      @map("created_by")
  updatedAt DateTime @updatedAt @map("updated_at")
  updatedBy Int      @map("updated_by")

  collaborators Collaborator[] @relation("CollaboratorProject")
  assets        Asset[]        @relation("AssetsProject")
  invites       Invite[]

  @@map("projects")
}

model Collaborator {
  id        Int       @id @default(autoincrement())
  uid       String    @unique @default(uuid(7)) @db.Uuid
  userId    Int       @map("user_id")
  projectId Int       @map("project_id")
  role      String    @default("viewer")
  startAt   DateTime? @map("start_at")
  endAt     DateTime? @map("end_at")
  status    Int       @default(1)
  createdAt DateTime  @default(now()) @map("created_at")
  createdBy Int       @map("created_by")
  updatedAt DateTime  @updatedAt @map("updated_at")
  updatedBy Int       @map("updated_by")

  project Project @relation("CollaboratorProject", fields: [projectId], references: [id])
  user    User    @relation("CollaboratorUser", fields: [userId], references: [id])

  @@unique([userId, projectId])
  @@map("collaborators")
}

model Asset {
  id        Int      @id @default(autoincrement())
  uid       String   @unique @default(uuid(7)) @db.Uuid
  type      String
  name      String
  ownerId   Int      @map("owner_id")
  projectId Int      @map("project_id")
  path      String
  configs   Json?
  binary    Bytes?
  text      String?
  status    Int      @default(1)
  createdAt DateTime @default(now()) @map("created_at")
  createdBy Int      @map("created_by")
  updatedAt DateTime @updatedAt @map("updated_at")
  updatedBy Int      @map("updated_by")

  project Project @relation("AssetsProject", fields: [projectId], references: [id])
  user    User?   @relation("AssetOwner", fields: [ownerId], references: [id])

  @@unique([projectId, path])
  @@map("assets")
}

model Invite {
  id        Int       @id @default(autoincrement())
  uid       String    @unique @default(uuid(7)) @db.Uuid
  token     String    @unique
  projectId Int       @map("project_id")
  project   Project   @relation(fields: [projectId], references: [id], onDelete: Cascade)
  createdBy Int       @map("created_by")
  creator   User      @relation("InviteCreator", fields: [createdBy], references: [id])
  role      String    @default("viewer")
  expiresAt DateTime? @map("expires_at")
  createdAt DateTime  @default(now()) @map("created_at")
  usedAt    DateTime? @map("used_at")
  usedBy    Int?      @map("used_by")
  user      User?     @relation("InviteUsedBy", fields: [usedBy], references: [id])

  @@map("invites")
}
